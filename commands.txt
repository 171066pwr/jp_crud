//create quarkus web project
mvn io.quarkus.platform:quarkus-maven-plugin:3.30.1:create -DprojectGroupId=com.mycompany.app -DprojectArtifactId=fryzjer -Dextensions=rest -DnoCode


//launch jar with overridden property value - i.e. quarkus.http.port
java -jar -Dquarkus.http.port=8090 fryzjer-dev.jar

//launch second instance with disabled db backup
java -jar -Dquarkus.http.port=8090 -Ddb.backup.enabled=false target\quarkus-app\quarkus-run.jar


java -jar -Ddb.backup.period=60s target\quarkus-app\quarkus-run.jar
